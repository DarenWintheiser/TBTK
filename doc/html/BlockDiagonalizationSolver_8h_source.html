164 inline const std::complex<double> BlockDiagonalizationSolver::getAmplitude(
165  int state,
166  const Index &index
167 ){
168  const Model &model = getModel();
169  unsigned int block = stateToBlockMap.at(state);
170  unsigned int offset = eigenVectorOffsets.at(block);
171  unsigned int linearIndex = model.getBasisIndex(index);
172  unsigned int firstStateInBlock = blockToStateMap.at(block);
173  unsigned int lastStateInBlock = firstStateInBlock + numStatesPerBlock.at(block)-1;
174  offset += (state - firstStateInBlock)*numStatesPerBlock.at(block);
175  if(linearIndex >= firstStateInBlock && linearIndex <= lastStateInBlock)
176  return eigenVectors[offset + (linearIndex - firstStateInBlock)];
177  else
178  return 0;
179 }

182  const Index &blockIndex,

184  const Index &intraBlockIndex
185 ){
186  int firstStateInBlock = getModel().getHoppingAmplitudeSet()->getFirstIndexInBlock(
187  blockIndex
188  );
189  unsigned int block = stateToBlockMap.at(firstStateInBlock);
190  TBTKAssert(
191  state >= 0 && state < (int)numStatesPerBlock.at(block),
192  "BlockDiagonalizationSolver::getAmplitude()",
193  "Out of bound error. The block with block Index "
194  << blockIndex.toString() << " has "
195  << numStatesPerBlock.at(block) << " states, but state "
196  << state << " was requested.",
197  ""
198  );
199  unsigned int offset = eigenVectorOffsets.at(block) + state*numStatesPerBlock.at(block);
200  unsigned int linearIndex = getModel().getBasisIndex(
201  Index(blockIndex, intraBlockIndex)
202  );
203 // Streams::out << linearIndex << "\t" << Index(blockIndex, intraBlockIndex).toString() << "\n";
204 
205  return eigenVectors[offset + (linearIndex - firstStateInBlock)];
206 }
