300 inline unsigned int HoppingAmplitudeTree::getSizeInBytes() const{
301  unsigned int size = 0;
302  for(unsigned int n = 0; n < hoppingAmplitudes.size(); n++)
303  size += hoppingAmplitudes[n].getSizeInBytes();
304  for(unsigned int n = 0; n < children.size(); n++)
305  size += children[n].getSizeInBytes();
306 
307  size += (
308  hoppingAmplitudes.capacity() - hoppingAmplitudes.size()
309  )*sizeof(HoppingAmplitude);
310 
311  size += (
312  children.capacity() - children.size()
313  )*sizeof(HoppingAmplitudeTree);
314 
315  return size + sizeof(HoppingAmplitudeTree);
316 }
