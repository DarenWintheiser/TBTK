283 inline unsigned int BitRegister::getNumOneBits() const{
284  unsigned int x = values;
285  x = x - ((x >> 1) & 0x55555555);
286  x = (x & 0x33333333) + ((x >> 2) & 0x33333333);
287  x = (x & 0x0F0F0F0F) + ((x >> 4) & 0x0F0F0F0F);
288  x = x + (x >> 8);
289  x = x + (x >> 16);
290  return (x & 0x0000003F);
291 }
